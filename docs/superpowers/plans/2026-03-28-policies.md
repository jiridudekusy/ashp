# ASHP Policies Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add hierarchical rule grouping ("policies") with per-agent assignment so each agent only sees rules from its assigned policies.

**Architecture:** New `policies`, `policy_children`, `agent_policies` tables in SQLite (v2→v3 migration). Server resolves policy tree per-agent and sends flat rule list via IPC. Go proxy switches from single rule list to per-agent rule map. GUI gets sidebar tree + detail panel layout.

**Tech Stack:** Node.js/Express (server), SQLite/better-sqlite3 (DB), Go (proxy), React 19 + CSS modules (GUI)

// Migration v2 -> v3: policies (hierarchical rule grouping)
if (user_version < 3) {
  db.transaction(() => {
    db.exec(`
      CREATE TABLE policies (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        name TEXT NOT NULL UNIQUE,
        description TEXT DEFAULT '',
        created_at DATETIME NOT NULL DEFAULT(datetime('now'))
      );

      CREATE TABLE policy_children (
        parent_id INTEGER NOT NULL REFERENCES policies(id) ON DELETE CASCADE,
        child_id INTEGER NOT NULL REFERENCES policies(id) ON DELETE CASCADE,
        PRIMARY KEY (parent_id, child_id),
        CHECK (parent_id != child_id)
      );

      CREATE TABLE agent_policies (
        agent_id INTEGER NOT NULL REFERENCES agents(id) ON DELETE CASCADE,
        policy_id INTEGER NOT NULL REFERENCES policies(id) ON DELETE CASCADE,
        PRIMARY KEY (agent_id, policy_id)
      );

      ALTER TABLE rules ADD COLUMN policy_id INTEGER REFERENCES policies(id) ON DELETE SET NULL;
    `);

    // Create default policy and assign all existing rules to it
    const info = db.prepare("INSERT INTO policies (name, description) VALUES ('default', 'Default policy for all rules')").run();
    db.prepare('UPDATE rules SET policy_id = ?').run(info.lastInsertRowid);

    db.pragma('user_version = 3');
  })();
}

```javascript
import { PoliciesDAO } from '../interfaces.js';

function deserialize(row) {
  if (!row) return null;
  return { ...row };
}

export class SqlitePoliciesDAO extends PoliciesDAO {
  #db;
  #stmts;

  constructor(db) {
    super();
    this.#db = db;
    this.#stmts = {
      list: db.prepare('SELECT * FROM policies ORDER BY name'),
      get: db.prepare('SELECT * FROM policies WHERE id = ?'),
      getByName: db.prepare('SELECT * FROM policies WHERE name = ?'),
      insert: db.prepare('INSERT INTO policies (name, description) VALUES (@name, @description)'),
      delete: db.prepare('DELETE FROM policies WHERE id = ?'),
      addChild: db.prepare('INSERT INTO policy_children (parent_id, child_id) VALUES (?, ?)'),
      removeChild: db.prepare('DELETE FROM policy_children WHERE parent_id = ? AND child_id = ?'),
      getChildren: db.prepare('SELECT p.* FROM policies p JOIN policy_children pc ON p.id = pc.child_id WHERE pc.parent_id = ?'),
      getParents: db.prepare('SELECT p.* FROM policies p JOIN policy_children pc ON p.id = pc.parent_id WHERE pc.child_id = ?'),
      assignAgent: db.prepare('INSERT OR IGNORE INTO agent_policies (agent_id, policy_id) VALUES (?, ?)'),
      unassignAgent: db.prepare('DELETE FROM agent_policies WHERE agent_id = ? AND policy_id = ?'),
      getAgentPolicies: db.prepare('SELECT p.* FROM policies p JOIN agent_policies ap ON p.id = ap.policy_id WHERE ap.agent_id = ?'),
      getPolicyAgents: db.prepare('SELECT a.id, a.name, a.description, a.enabled FROM agents a JOIN agent_policies ap ON a.id = ap.agent_id WHERE ap.policy_id = ?'),
      ruleCount: db.prepare('SELECT COUNT(*) as count FROM rules WHERE policy_id = ?'),
      checkCycle: db.prepare(`
        WITH RECURSIVE ancestors(id) AS (
          SELECT parent_id FROM policy_children WHERE child_id = ?
          UNION ALL
          SELECT pc.parent_id FROM policy_children pc JOIN ancestors a ON pc.child_id = a.id
        )
        SELECT 1 FROM ancestors WHERE id = ?
      `),
      resolveDescendants: db.prepare(`
        WITH RECURSIVE descendants(id) AS (
          SELECT child_id FROM policy_children WHERE parent_id IN (SELECT policy_id FROM agent_policies WHERE agent_id = ?)
          UNION
          SELECT policy_id FROM agent_policies WHERE agent_id = ?
          UNION ALL
          SELECT pc.child_id FROM policy_children pc JOIN descendants d ON pc.parent_id = d.id
        )
        SELECT DISTINCT id FROM descendants
      `),
      rulesByPolicies: db.prepare(`
        SELECT r.* FROM rules r WHERE r.policy_id IN (
          WITH RECURSIVE descendants(id) AS (
            SELECT policy_id FROM agent_policies WHERE agent_id = ?
            UNION ALL
            SELECT pc.child_id FROM policy_children pc JOIN descendants d ON pc.parent_id = d.id
          )
          SELECT DISTINCT id FROM descendants
        ) AND r.enabled = 1
        ORDER BY r.priority DESC
      `),
    };
  }

  async list() {
    const policies = this.#stmts.list.all().map(deserialize);
    for (const p of policies) {
      p.rule_count = this.#stmts.ruleCount.get(p.id).count;
    }
    return policies;
  }

  async get(id) {
    const policy = deserialize(this.#stmts.get.get(id));
    if (!policy) return null;
    policy.children = this.#stmts.getChildren.all(id).map(deserialize);
    policy.parents = this.#stmts.getParents.all(id).map(deserialize);
    policy.agents = this.#stmts.getPolicyAgents.all(id);
    policy.rule_count = this.#stmts.ruleCount.get(id).count;
    return policy;
  }

  async create({ name, description = '' }) {
    const info = this.#stmts.insert.run({ name, description });
    return this.get(info.lastInsertRowid);
  }

  async update(id, changes) {
    if (!this.#stmts.get.get(id)) return null;
    const fields = [];
    const params = { id };
    for (const [k, v] of Object.entries(changes)) {
      if (k === 'id') continue;
      params[k] = v;
      fields.push(`${k} = @${k}`);
    }
    if (fields.length) {
      this.#db.prepare(`UPDATE policies SET ${fields.join(',')} WHERE id = @id`).run(params);
    }
    return this.get(id);
  }

  async delete(id) {
    this.#stmts.delete.run(id);
  }

  async addChild(parentId, childId) {
    if (parentId === childId) throw new Error('Cannot add policy as its own child (cycle)');
    const cycle = this.#stmts.checkCycle.get(parentId, childId);
    if (cycle) throw new Error('Adding this child would create a cycle');
    this.#stmts.addChild.run(parentId, childId);
  }

  async removeChild(parentId, childId) {
    this.#stmts.removeChild.run(parentId, childId);
  }

  async getTree() {
    const all = this.#stmts.list.all().map(deserialize);
    for (const p of all) {
      p.children = this.#stmts.getChildren.all(p.id).map(deserialize);
      p.rule_count = this.#stmts.ruleCount.get(p.id).count;
    }
    // Build tree: policies without parents are roots
    const childIds = new Set();
    for (const p of all) {
      for (const c of p.children) childIds.add(c.id);
    }
    return all.filter(p => !childIds.has(p.id));
  }

  async assignToAgent(policyId, agentId) {
    this.#stmts.assignAgent.run(agentId, policyId);
  }

  async unassignFromAgent(policyId, agentId) {
    this.#stmts.unassignAgent.run(agentId, policyId);
  }

  async getAgentPolicies(agentId) {
    return this.#stmts.getAgentPolicies.all(agentId).map(deserialize);
  }

  async resolveAgentRules(agentId) {
    return this.#stmts.rulesByPolicies.all(agentId).map(row => ({
      ...row,
      methods: JSON.parse(row.methods),
      enabled: !!row.enabled,
    }));
  }
}
```

```go
type Evaluator struct {
	mu      sync.RWMutex
	byAgent map[string][]compiledRule
}

func NewEvaluator() *Evaluator {
	return &Evaluator{byAgent: make(map[string][]compiledRule)}
}

// LoadMap replaces the rule set with a per-agent map.
func (e *Evaluator) LoadMap(agentRules map[string][]Rule) {
	byAgent := make(map[string][]compiledRule, len(agentRules))
	for agent, rules := range agentRules {
		compiled := make([]compiledRule, 0, len(rules))
		for _, r := range rules {
			if !r.Enabled { continue }
			re, err := regexp.Compile(r.URLPattern)
			if err != nil { continue }
			compiled = append(compiled, compiledRule{Rule: r, re: re})
		}
		sort.Slice(compiled, func(i, j int) bool {
			return compiled[i].Priority > compiled[j].Priority
		})
		byAgent[agent] = compiled
	}
	e.mu.Lock()
	e.byAgent = byAgent
	e.mu.Unlock()
}

// Load replaces with a single flat list (backwards compat during transition).
func (e *Evaluator) Load(rules []Rule) {
	e.LoadMap(map[string][]Rule{"__all__": rules})
}

// Match finds the highest-priority rule for the given agent.
func (e *Evaluator) Match(agentID, url, method string) *Rule {
	e.mu.RLock()
	defer e.mu.RUnlock()
	rules := e.byAgent[agentID]
	for _, cr := range rules {
		if !cr.re.MatchString(url) { continue }
		if len(cr.Methods) > 0 {
			found := false
			for _, m := range cr.Methods {
				if m == method { found = true; break }
			}
			if !found { continue }
		}
		r := cr.Rule
		return &r
	}
	return nil
}
```
